Xin lỗi bạn, mình sẽ bổ sung chi tiết toàn bộ giao diện theo đúng yêu cầu. Sau đây là toàn bộ code cho giao diện, bao gồm cả cấu trúc React Native và CSS:

File Search.js

javascript
import React, { useState, useEffect } from "react";
import {
  View,
  Text,
  TextInput,
  FlatList,
  TouchableOpacity,
  ScrollView,
  Image,
  ActivityIndicator,
  Keyboard,
} from "react-native";
import axios from "axios";
import styles from "./Search_css";

const Search = () => {
  const [currentView, setCurrentView] = useState("default"); // Chế độ hiển thị ("default", "searching", "results")
  const [searchText, setSearchText] = useState(""); // Nội dung tìm kiếm
  const [images, setImages] = useState([]); // Kết quả tìm kiếm
  const [isLoading, setIsLoading] = useState(false); // Trạng thái đang tải
  const [noResults, setNoResults] = useState(false); // Không có kết quả

  // Xử lý tìm kiếm
  const handleSearch = async () => {
    if (!searchText.trim()) return;
    setIsLoading(true);
    setCurrentView("results"); // Hiển thị trang kết quả tìm kiếm
    try {
      const response = await axios.post("http://192.168.0.101:5001/api/search_image", {
        keyword: searchText,
      });

      const fetchedImages = response.data.best_image_urls;
      if (fetchedImages.length > 0) {
        setImages(fetchedImages); // Gán ảnh vào state
        setNoResults(false);
      } else {
        setImages([]);
        setNoResults(true); // Không có kết quả
      }
    } catch (error) {
      console.error("Error fetching images:", error);
      setImages([]);
      setNoResults(true);
    } finally {
      setIsLoading(false);
    }
  };

  // Đổi trạng thái về mặc định
  const handleCancel = () => {
    setCurrentView("default");
    setSearchText("");
    setImages([]);
    setNoResults(false);
    Keyboard.dismiss();
  };

  return (
    <View style={styles.container}>
      {/* Header */}
      <View style={styles.header}>
        <TextInput
          style={styles.searchBar}
          placeholder="Tìm kiếm ý tưởng"
          value={searchText}
          onFocus={() => setCurrentView("searching")}
          onChangeText={(text) => setSearchText(text)}
          onSubmitEditing={handleSearch}
          returnKeyType="search"
        />
        {currentView === "searching" && (
          <TouchableOpacity onPress={handleCancel} style={styles.cancelButton}>
            <Text style={styles.cancelText}>Hủy</Text>
          </TouchableOpacity>
        )}
      </View>

      {/* Body */}
      <View style={styles.body}>
        {currentView === "default" && (
          <ScrollView>
            {/* Nội dung mặc định */}
            <Text style={styles.defaultText}>
              Đây là nội dung trang chính, hiển thị gợi ý theo yêu cầu.
            </Text>
          </ScrollView>
        )}
        {currentView === "searching" && (
          <ScrollView>
            {/* Hiển thị lịch sử tìm kiếm */}
            <Text style={styles.historyTitle}>Lịch sử tìm kiếm</Text>
            <TouchableOpacity>
              <Text style={styles.historyItem}>Câu nói tâm trạng</Text>
            </TouchableOpacity>
            <TouchableOpacity>
              <Text style={styles.historyItem}>Ảnh nền đẹp</Text>
            </TouchableOpacity>
          </ScrollView>
        )}
        {currentView === "results" && (
          <>
            {isLoading ? (
              <ActivityIndicator size="large" color="#007AFF" />
            ) : noResults ? (
              <Text style={styles.noResultsText}>Không có ảnh nào.</Text>
            ) : (
              <FlatList
                data={images}
                keyExtractor={(item, index) => index.toString()}
                numColumns={3}
                renderItem={({ item }) => (
                  <Image
                    source={{ uri: item }}
                    style={styles.imageStyle}
                  />
                )}
                contentContainerStyle={styles.imageList}
              />
            )}
          </>
        )}
      </View>

      {/* Footer */}
      <View style={styles.footer}>
        <Text>Footer - Chứa điều hướng</Text>
      </View>
    </View>
  );
};

export default Search;
